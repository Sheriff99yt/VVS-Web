}
public enum SensorStatus {
    OK,
    WARN,
    FAIL
}
public class Sensor : Machine {
        public List<float> Readings = new List<float>();
        public SensorStatus Status = SensorStatus.OK;
        public Machine Host = null;
        public Dictionary<string, string> Tags = 0;
    public async void Sample() {
        // Sample
        this.Readings.Add(this.Power);
        Console.WriteLine("Sampled");
    }
    public override void Report() {
        // Report
        Console.WriteLine("--- Readings ---");
        foreach (var val in this.Readings) {
            Console.WriteLine(val.ToString());
        }
    }
    public void on_tick() {
        Console.WriteLine("Tick");
    }
    public void on_start() {
        switch (this.Status) {
            case SensorStatus.OK:
            this.Sample();
            this.Report();
                    break;
            case SensorStatus.WARN:
            Console.WriteLine("Warn");
            this.Report();
                    break;
            case SensorStatus.FAIL:
            Console.WriteLine("Fail");
            this.Report();
                    break;
        }
    }
}
