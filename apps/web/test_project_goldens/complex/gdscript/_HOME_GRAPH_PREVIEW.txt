enum SensorStatus { OK, WARN, FAIL }
class_name Sensor
extends Machine
    var Readings = []
    var Status = SensorStatus.OK
    var Host = null
    var Tags = 0
    func Sample() -> void:
        # Sample
        self.Readings.append(self.Power)
        print("Sampled")
    func Report() -> void:
        # Report
        print("--- Readings ---")
        for val in self.Readings:
            print(str(val))
    func on_tick() -> void:
        print("Tick")
    func on_start() -> void:
        _vvs_sel = self.Status
        if _vvs_sel == SensorStatus.OK:
            self.Sample()
            self.Report()
        elif _vvs_sel == SensorStatus.WARN:
            print("Warn")
            self.Report()
        elif _vvs_sel == SensorStatus.FAIL:
            print("Fail")
            self.Report()
