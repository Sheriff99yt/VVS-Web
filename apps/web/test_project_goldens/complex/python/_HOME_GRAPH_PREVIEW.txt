class SensorStatus(Enum):
    OK = 1
    WARN = 2
    FAIL = 3
class Sensor(Machine):
    Readings = []
    Status = SensorStatus.OK
    Host = None
    Tags = 0
    async def Sample(self):
        # Sample
        self.Readings.append(self.Power)
        print("Sampled")
    def Report(self):
        # Report
        print("--- Readings ---")
        for val in self.Readings:
            print(str(val))
    def on_tick(self):
        print("Tick")
    def on_start(self):
        _vvs_sel = self.Status
        if _vvs_sel == SensorStatus.OK:
            self.Sample()
            self.Report()
        elif _vvs_sel == SensorStatus.WARN:
            print("Warn")
            self.Report()
        elif _vvs_sel == SensorStatus.FAIL:
            print("Fail")
            self.Report()
