const SensorStatus = Object.freeze({
  OK: 1,
  WARN: 2,
  FAIL: 3
});
class Sensor extends Machine {
    Readings = [];
    Status = SensorStatus.OK;
    Host = null;
    Tags = 0;
  async Sample() {
    // Sample
    this.Readings.push(this.Power);
    console.log("Sampled");
  }
  Report() {
    // Report
    console.log("--- Readings ---");
    for (const val of this.Readings) {
        console.log(String(val));
    };
  }
  on_tick() {
      console.log("Tick");
  }
  on_start() {
      switch (this.Status) {
        case SensorStatus.OK:
          this.Sample();
          this.Report();
              break;
        case SensorStatus.WARN:
          console.log("Warn");
          this.Report();
              break;
        case SensorStatus.FAIL:
          console.log("Fail");
          this.Report();
              break;
      };
  }
}
