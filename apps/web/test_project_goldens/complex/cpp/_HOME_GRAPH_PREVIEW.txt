enum class SensorStatus {
    OK,
    WARN,
    FAIL
};
class Sensor : public Machine {
public:
        std::vector<float> Readings = {};
        SensorStatus Status = SensorStatus::OK;
        Machine Host = {};
        std::unordered_map<std::string, std::string> Tags = 0;
    void Sample() {
        // Sample
        Readings.push_back(Power);
        std::cout << "Sampled" << std::endl;
    }
    void Report() {
        // Report
        std::cout << "--- Readings ---" << std::endl;
        for (float val : Readings) {
            std::cout << std::to_string(val) << std::endl;
        }
    }
    void on_tick() {
        std::cout << "Tick" << std::endl;
    }
    virtual void on_start() override {
        switch (Status) {
          case SensorStatus::OK:
            Sample();
            Report();
                break;
          case SensorStatus::WARN:
            std::cout << "Warn" << std::endl;
            Report();
                break;
          case SensorStatus::FAIL:
            std::cout << "Fail" << std::endl;
            Report();
                break;
        }
    }
};
