pub enum SensorStatus {
    OK,
    WARN,
    FAIL
}
pub struct Sensor {
    base: Machine,

    pub Readings: Vec<f32>,
    pub Status: SensorStatus,
    pub Host: Machine,
    pub Tags: HashMap<String, String>,
}
impl Sensor {

    // (x) Declare Sample
    pub async fn Sample(&mut self) {
        self.Readings.push(self.Power);
        println!("{}", "Sampled");
    }
    // (x) Declare Report
    pub fn Report(&mut self) {
        println!("{}", "--- Readings ---");
        for val in self.Readings.iter() {
            println!("{}", val.to_string());
        }
    }
    pub fn on_tick(&mut self) {
        println!("{}", "Tick");
    }
    pub fn on_start(&mut self) {
        let _vvs_sel = self.Status;
        if _vvs_sel == SensorStatus::OK {
            self.Sample();
            self.Report();
        } else if _vvs_sel == SensorStatus::WARN {
            println!("{}", "Warn");
            self.Report();
        } else if _vvs_sel == SensorStatus::FAIL {
            println!("{}", "Fail");
            self.Report();
        }
    }
}
